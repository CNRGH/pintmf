loss <- sapply(R_bxd, function (rr){
  W <- rr$W
  l <- sapply(1:length(rr$H), function (ii) {
    h <- rr$H[[ii]]
    y <- bxd_filtered[[ii]]
    sum((y-W%*%h)^2)/(nrow(W)*nrow(y))
  })
  mean(l)
})
loss_W <- sapply(R_bxd, function (rr){
  W <- rr$W
  sum(W^2)/ncol(W)
})
